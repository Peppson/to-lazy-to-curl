icon helper:
        var icons = Enum.GetValues(typeof(FontAwesomeIcon))
                        .Cast<FontAwesomeIcon>()
                        .Where(t => t != FontAwesomeIcon.None)
                        .Where(icon => typeof(FontAwesomeIcon)
                                        .GetMember(icon.ToString())
                                        .FirstOrDefault() != null);

        string desktopPath = Environment.GetFolderPath(Environment.SpecialFolder.Desktop);
        string filePath = Path.Combine(desktopPath, "FontAwesomeIcons.txt");
        File.WriteAllLines(filePath, icons.Select(i => i.ToString()));

        Console.WriteLine($"Saved {icons.Count()} icons to {filePath}");
